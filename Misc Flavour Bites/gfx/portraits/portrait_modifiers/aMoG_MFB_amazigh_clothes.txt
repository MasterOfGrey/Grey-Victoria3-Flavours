# Appropriate Arabic clothing for Amazigh/Berber culture now that the heritage is African instead of Middle-Eastern
mog_suitable_berber_clothes = {
	usage = game
	selection_behavior = weighted_random
	
	# Poor clothes may overlap with arabic_poor_clothes definition?
	arabic_clothes_without_middle_eastern_heritage = {
		dna_modifiers = {
			accessory = {
				mode = add
				gene = outfits
				template = arabic_outfits
				range = { 0 1 }		# For the randomness to work correctly
			}
		}
		weight = {
			base = 0
			modifier = {
				add = 1000
				OR = {
					AND = { # For Characters
						arabic_clothes_trigger = no											# Avoid nudity if trigger is overwritten
						arabic_clothes_without_middle_eastern_heritage_trigger = yes
						agitator_character_trigger = no
						armed_forces_clothes_trigger = no
					}
					AND = {	# For Pops
						arabic_clothes_pop_trigger = no										# Avoid nudity if trigger is overwritten
						arabic_clothes_without_middle_eastern_heritage_pop_trigger = yes
						OR = {
							poor_clothes_pop_trigger = yes
							middle_clothes_pop_trigger = yes
						}
					}
				}
			}
		}
	}
	arabic_upper_clothes_without_middle_eastern_heritage = {
		dna_modifiers = {
			accessory = {
				mode = add
				gene = outfits
				template = arabic_upper_outfits
				range = { 0 1 }		# For the randomness to work correctly
			}
		}
		weight = {
			base = 0
			modifier = {
				add = 1000
				OR = {
					AND = { # For Characters
						arabic_clothes_trigger = no											# Avoid nudity if trigger is overwritten
						arabic_clothes_without_middle_eastern_heritage_trigger = yes
					}
					AND = {	# For Pops
						arabic_clothes_pop_trigger = no										# Avoid nudity if trigger is overwritten
						upper_clothes_pop_trigger = yes
						arabic_clothes_without_middle_eastern_heritage_pop_trigger = yes
					}
				}
			}
		}
	}
	
	arabic_coats_for_berber = {
		dna_modifiers = {
			accessory = {
				mode = add
				gene = coats
				template = arabic_coats
				range = { 0 1 } # For the randomness to work correctly
			}
		}
		weight = {
			base = 0
			modifier = {
				add = 500
				AND = {
					OR = {
						arabic_clothes_without_middle_eastern_heritage_trigger = yes
						arabic_clothes_without_middle_eastern_heritage_pop_trigger = yes
					}
					OR = {
						military_clothes_trigger = yes
						officers_clothes_pop_trigger = yes
						AND = {
							armed_forces_clothes_trigger = yes
							OR = {
								politician_clothes_trigger = yes
								ruler_clothes_trigger = yes
							}
						}
					}
				}
			}
		}
	}
	
	# Appropriate weighting carried across from those using `has_discrimination_trait = middle_eastern_heritage`
	# Fancy Landowners
	berber_landowners_clothes = {
		dna_modifiers = {
			accessory = {
				mode = add
				gene = outfits
				template = male_outfit_arabic_royal_01
				range = { 0 1 } # For the randomness to work correctly
			}
			accessory = {
				mode = add
				gene = legwear
				template = slavic_legwear
				range = { 0 1 } # For the randomness to work correctly
			}
			accessory = {
				mode = add
				gene = belts
				template = no_belt
				range = { 0 1 } # For the randomness to work correctly
			}
		}
		weight = {
			base = 0
			modifier = {
				add = 1000
				OR = {
					scope:pop ?= {
                        is_pop_type = aristocrats
                        culture = cu:berber
                    }
					scope:character ?= {
						is_interest_group_type = ig_landowners
						age >= define:NCharacters|ADULT_AGE
						OR = {
							has_role = politician
							AND = {
								has_role = executive
								NOT = {
									has_dlc_feature = mp1_cosmetics
								}
							}
						}
						NAND = {    # monarchs wear other stuff
							is_ruler = yes
							owner ?={
								has_law = law_type:law_monarchy
							}
						}
						culture = cu:berber
					}
					scope:character ?= {
						is_ruler = yes
						age >= define:NCharacters|ADULT_AGE
						culture = cu:berber
					}
				}
			}
		}
	}
	berber_landowners_headgear = {
		dna_modifiers = {
			accessory = {
				mode = add
				gene = headgear
				template = arabic_landowners_hat
				range = { 0 1 } # For the randomness to work correctly
			}
		}
		weight = {
			base = 0
			modifier = {
				add = 5000
				
				has_dlc_feature = ep1_cosmetics
				
				OR = {
					scope:pop ?= {
						is_pop_type = aristocrats
						culture = cu:berber
					}
					scope:character ?= {
						is_interest_group_type = ig_landowners
						age > 12
						has_role = politician
						NAND = {    # monarchs never wear this hat
							is_ruler = yes
							owner ?={
								has_law = law_type:law_monarchy
							}
						}
						culture = cu:berber
					}
				}
			}
		}
	}
	# Fez
	berber_military_fez = { # Excludes the fancy female headgear meant for high class pops
		dna_modifiers = {
			accessory = {
				mode = add
				gene = headgear
				template = arabic_military_fez
				range = { 0 1 } # For the randomness to work correctly
			}
		}
		weight = {
			base = 0
			modifier = {
				add = 200
				scope:pop ?= {
					AND = {
						culture = cu:berber
						OR = {
							home_country = {
								exists = c:EGY
								c:EGY = this
								c:EGY = { has_law = law_type:law_monarchy }
							}
							home_country = {
								exists = c:TUR
								c:TUR = this
								c:TUR = { has_law = law_type:law_monarchy }
							}
							home_country = {
								exists = c:TUN
								c:TUN = this
								c:TUN = { has_law = law_type:law_monarchy }
							}
							home_country = {
								exists = c:MOR
								c:MOR = this
								c:MOR = { has_law = law_type:law_monarchy }
							}
						}
					}
				}
				soldier_clothes_pop_trigger = yes
			}
		}
	}
	# Agitator hats
	berber_agitator_headgear = {
		dna_modifiers = {
			accessory = {
				mode = add
				gene = headgear
				template = arabic_headgear_agitators
				range = { 0 1 } # For the randomness to work correctly
			}
		}
		weight = {
			base = 0
			modifier = {
				add = 1000
				scope:character ?= {
					culture = cu:berber
				}
				agitator_character_trigger = yes
			}
		}
	}
}
