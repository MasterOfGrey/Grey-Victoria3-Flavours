# Generic
ai_strategy_unify_sokoto = {	# Based off ethiopia
	icon = "gfx/interface/icons/ai_strategy_icons/unification.dds"

	type = diplomatic

	will_form_power_bloc = {
		always = yes
	}

	undesirable_infamy_level = {
		value = 50
	}

	unacceptable_infamy_level = {
		value = 100
	}

	ideological_opinion_effect_mult = 0.5

	wargoal_maneuvers_fraction = {
		value = 0.2
	}

	diplomatic_play_neutrality = {
		value = 0
	}

	diplomatic_play_boldness = {
		value = 50
	}

	obligation_value = {
		value = 0
	}

	recklessness = {
		value = 1.0
	}

	aggression = {
		value = 0.5

		if = {
			limit = {
				scope:target_country = {
					any_scope_state = {
						OR = {
							state_region = s:STATE_HAUSALAND		# Necessary
							state_region = s:STATE_OUTER_HAUSALAND	# Necessary
							state_region = s:STATE_EAST_HAUSALAND	# Necessary
							state_region = s:STATE_NIGERIA			# Necessary
							state_region = s:STATE_BORNU	# Bonus
							state_region = s:STATE_NIGER	# Bonus
						}
					}
				}
			}
			add = 10.0
		}

		# Less aggressive against countries we're just antagonistic against, rather than belligerent/domineering
		if = {
			limit = {
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}
			}
			multiply = 0.25
		}
	}

	building_group_weights = {
		bg_army = 1.5
	}

	state_value = {
		if = {
			limit = {
				scope:target_state = {
					state_region = {
						OR = {
							is_homeland = cu:hausa
							is_homeland = cu:fulbe
						}
					}
				}
				can_reach_target_state = yes
			}
			add = 100
		}
	}

	wargoal_scores = {
		annex_country = {
			if = {
				limit = {
					scope:target_country = {
						any_scope_state = {
							OR = {
								state_region = s:STATE_HAUSALAND		# Necessary
								state_region = s:STATE_OUTER_HAUSALAND	# Necessary
								state_region = s:STATE_EAST_HAUSALAND	# Necessary
								state_region = s:STATE_NIGERIA			# Necessary
								state_region = s:STATE_BORNU	# Bonus
								state_region = s:STATE_NIGER	# Bonus
							}
						}
					}
					can_reach_target_country = yes
				}
				add = 300
			}
		}
	}

	secret_goal_scores = {
		befriend = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				scope:target_country = {
					any_scope_state = {
						OR = {
							state_region = s:STATE_HAUSALAND		# Necessary
							state_region = s:STATE_OUTER_HAUSALAND	# Necessary
							state_region = s:STATE_EAST_HAUSALAND	# Necessary
							state_region = s:STATE_NIGERIA			# Necessary
							state_region = s:STATE_BORNU	# Bonus
							state_region = s:STATE_NIGER	# Bonus
						}
					}
				}
			}
			score = {
				value = 150

				if = {
					limit = {
						can_reach_target_country = no
					}
					add = 150
				}
			}
		}
		conquer = {
			trigger = {
				NOT = { scope:target_country = { is_subject_of = root } }
				scope:target_country = {
					any_scope_state = {
						OR = {
							state_region = s:STATE_HAUSALAND		# Necessary
							state_region = s:STATE_OUTER_HAUSALAND	# Necessary
							state_region = s:STATE_EAST_HAUSALAND	# Necessary
							state_region = s:STATE_NIGERIA			# Necessary
							state_region = s:STATE_BORNU	# Bonus
							state_region = s:STATE_NIGER	# Bonus
						}
					}
				}
				can_reach_target_country = yes
			}
			score = {
				value = 500
			}
		}
	}

	possible = {
		c:SOK ?= THIS
		NOT = { has_variable = sokoto_unity }
	}

	weight = {
		value = 25

		add = {
			value = army_size
			multiply = 10
		}

		if = {
			limit = { NOT = { is_country_type = unrecognized } }
			multiply = 10.0
		}
		else_if = {
			limit = { country_rank = rank_value:unrecognized_major_power }
			multiply = 5.0
		}
		else_if = {
			limit = { country_rank = rank_value:unrecognized_regional_power }
			multiply = 3.0
		}

		# Sokoto less equipped to jihad without centralization (which enables the JE)
		if = {
			limit = { NOT = { has_journal_entry = je_sokotan_unification } }
			multiply = 0.5
		}
	}
}
