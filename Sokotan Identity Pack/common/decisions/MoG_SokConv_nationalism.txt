sokoto_unity_decision = {
	is_shown = {
		exists = c:SOK
		c:SOK = THIS
		NOT = { has_variable = sokoto_unity }
	}

	possible = {
		is_at_war = no
		has_technology_researched = nationalism
		c:SOK ?= {
			AND = {
				owns_entire_state_region = STATE_HAUSALAND
				owns_entire_state_region = STATE_OUTER_HAUSALAND
				owns_entire_state_region = STATE_EAST_HAUSALAND
				owns_entire_state_region = STATE_NIGERIA
				owns_entire_state_region = STATE_NIGER
			}
		}
	}

	when_taken = {
		trigger_event = { id = sokoto_conversion.1 days = 0 popup = yes }
	}

	ai_chance = {
		base = 100
	}
}

# Local incorporations
sokoto_yoruba = {
	is_shown = {
		exists = c:SOK
		c:SOK = THIS
		has_variable = sokoto_unity
		NOT = { has_variable = yoruba_is_sokotan }
	}

	possible = {
		is_at_war = no
		c:SOK ?= {
			owns_entire_state_region = STATE_HAUSALAND
			owns_entire_state_region = STATE_YORUBA_STATES
			owns_entire_state_region = STATE_DAHOMEY
			owns_entire_state_region = STATE_BENIN
		}
		s:STATE_HAUSALAND.region_state:SOK = {
			is_incorporated = yes
		}
		s:STATE_YORUBA_STATES.region_state:SOK = {
			is_incorporated = yes
		}
		s:STATE_DAHOMEY.region_state:SOK = {
			is_incorporated = yes
		}
		s:STATE_BENIN.region_state:SOK = {
			is_incorporated = yes
		}
	}

	when_taken = {
		set_variable = yoruba_is_sokotan
		cu:yoruba = {
			save_scope_as = integrating_sokotan_culture
		}
		trigger_event = { id = sokoto_conversion.2 days = 0 popup = yes }
	}

	ai_chance = {
		base = 100
	}
}
