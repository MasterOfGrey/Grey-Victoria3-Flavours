	is_shown = {
		exists = c:RWD
		c:RWD = THIS
		NOT = { has_variable = ruanda_unity }
	}

		}
	}

	when_taken = {
		trigger_event = { id = ruanda_conversion.1 days = 0 popup = yes }
	}

	ai_chance = {
		base = 100
	}
}
